\begin{frame}{DRL algorithms: PPO - components}
  \twocol{
    \begin{itemize}
      \item \alert{rollout buffer}: A rollout is a sequence
      of iteractions between an agent and the environment \textbf{collected
      under the same current policy}. It can be imagined as \textit{watching
      the agent playing the current game} and recording what is does. 
      Its usage is a common practice with PPO algorithms according to
      \href{https://stable-baselines3.readthedocs.io/en/master/modules/ppo.html\#notes}{Stable-Baselines3}.
    \end{itemize}
  }{
    \begin{itemize}
      \item separate \textbf{actor} and \textbf{critic} networks, separate
        Adam optimizers. 
      \item Shared CNN backbones as seen before, with the difference that the critic's value head
        outputs a single $V(s)$.
      \item independent gradient-clip norms.
      \item both the learning rate and gradient norms are parametrized as:
      \texttt{max\_actor\_grad\_norm}, \texttt{max\_critic\_grad\_norm}, \texttt{actor\_learning\_rate},
      \texttt{critic\_learning\_rate}.
    \end{itemize}
  }
\end{frame}

\begin{frame}{Generalized Advantage Estimation \parencite{schulman2016gae}}
  \[
    \delta_t = r_t + \gamma\,(1-d_t)\,V(s_{t+1}) - V(s_t)
    \qquad
    A_t = \delta_t + \gamma\lambda\,(1-d_t)\,A_{t+1}
  \]
  \[
    \text{value target}_t = V(s_t) + A_t
  \]
  \begin{itemize}
    \item $\gamma = 0.95$, $\lambda = 0.95$ — computed backwards over the
      $1024$-step rollout;
    \item advantages are \alert{normalized} (zero mean, unit std) per
      minibatch before use, standard PPO variance-reduction trick;
    \item bootstraps are cut at terminal transitions ($d_t=1$), same as DQN.
  \end{itemize}
\end{frame}

\begin{frame}{Clipped Surrogate Objective}
  \[
    r_t(\theta) = \exp\!\big(\log \pi_\theta(a_t\mid s_t) - \log \pi_{\theta_{\text{old}}}(a_t\mid s_t)\big)
  \]
  \[
    \mathcal{L}^{\text{CLIP}}(\theta) = \mathbb{E}_t\Big[\min\big(r_t(\theta)A_t,\ \text{clip}(r_t(\theta), 1{-}\epsilon, 1{+}\epsilon)A_t\big)\Big]
  \]
  \[
    \mathcal{L}^{\text{actor}} = -\mathcal{L}^{\text{CLIP}} - c_{\text{ent}}\,\mathcal{H}[\pi_\theta]
    \qquad
    \mathcal{L}^{\text{critic}} = \big(V_\phi(s_t) - \text{value target}_t\big)^2
  \]
  \begin{itemize}
    \item $\epsilon = 0.1$ clip range — conservative, keeps policy updates
      small on a discrete action space of size $H{\cdot}W$;
    \item entropy bonus $c_{\text{ent}} = 0.005$ discourages premature
      convergence to a deterministic (over-exploiting) policy;
    \item actor and critic losses are backpropagated and clipped
      \alert{independently} through their own optimizers.
  \end{itemize}
\end{frame}
