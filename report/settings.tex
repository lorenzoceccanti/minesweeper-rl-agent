% Full-bleed figure with caption, sized to a fraction of textwidth:

}

% Two-column slide split: \twocol{left content}{right content}
\newcommand{\twocol}[2]{%

    \end{column}
    \begin{column}{0.48\textwidth}
      #2
    \end{column}
  \end{columns}
}
