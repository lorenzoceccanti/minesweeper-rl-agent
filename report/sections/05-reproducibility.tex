\section{Reproducibility \& Tooling}

\begin{frame}{A Single Entry Point: \texttt{main.py}}
  \twocol{
    \textbf{Three subcommands}
    \begin{itemize}
      \item \texttt{train} — train an agent, then \alert{auto-runs} a held-out
        test on the resulting checkpoint;
      \item \texttt{test} — evaluate an existing checkpoint;
      \item \texttt{game} — play/watch an episode interactively (pygame).
    \end{itemize}
  }{
    \textbf{Two-phase argument parsing}
    \begin{itemize}
      \item phase 1: bootstrap flags select \texttt{algorithm} (\texttt{dqn}/\texttt{ppo})
        and \texttt{mode};
      \item phase 2: remaining CLI flags are generated \alert{dynamically}
        from the matching subtree of \texttt{config.yaml} — no flag is
        hand-declared per algorithm.
    \end{itemize}
  }
\end{frame}

\begin{frame}{Configuration Priority}
  \centering
  \begin{tikzpicture}[node distance=0.6cm]
    \node[component, fill=lightgray] (defaults) {\texttt{config.yaml}\\defaults};
    \node[component, right=of defaults] (file) {\texttt{-{}-config}\\override file};
    \node[component, right=of file, fill=unipiblue!15] (cli) {CLI flags\\(\texttt{-{}-lr}, \texttt{-{}-episodes}, \ldots)};
    \node[component, right=of cli, fill=goodgreen!20] (merged) {merged\\config (deep merge)};
    \draw[flowarrow] (defaults) -- (file);
    \draw[flowarrow] (file) -- (cli);
    \draw[flowarrow] (cli) -- (merged);
  \end{tikzpicture}
  \vspace{0.3cm}
  Later stages \alert{win} on conflicting keys — a CLI flag always beats a
  config file, which always beats the shipped defaults. Every run is fully
  determined by the merged config, which is itself saved into the
  checkpoint for exact reproducibility.
\end{frame}

\begin{frame}{Experiment Tracking with Weights \& Biases}
  \twocol{
    \textbf{\texttt{tracking/wandb\_logger.py}}
    \begin{itemize}
      \item \texttt{log\_run()} during training: episode return, length,
        loss, \alert{cumulative} and \alert{rolling (100-episode)} win rate;
      \item \texttt{log\_test\_run()} after a held-out evaluation: win
        rate, mean return/length with std.
    \end{itemize}
  }{
    \textbf{Linking train $\leftrightarrow$ test}
    \begin{itemize}
      \item each checkpoint stores its \texttt{wandb\_run\_id};
      \item \texttt{test} \alert{resumes} that same W\&B run instead of
        opening a new one — training curves and held-out results live
        in one place per checkpoint;
      \item repeated test evaluations of the same checkpoint are logged
        as indexed rows, not overwritten.
    \end{itemize}
  }
\end{frame}
